//Find Second Large Number
#include <stdio.h>
int main()
{
    int a[100],n,max,smax,i;
    printf("Enter n value");
    scanf("%d",&n);
    printf("Enter %d Numbers",n);
    for(i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    max=smax=-9999;
    for(i=0;i<n;i++)
    {
        if(a[i]>max)
        {
            smax=max;
            max=a[i];
        }
        else if(a[i]>smax && a[i]!=max)
        {
            smax=a[i];
        }
    }
    printf("Second Largest Number %d",smax);
    return 0;
}
